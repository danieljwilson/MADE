import hddm
import numpy as np

#params#
num_subs = 30
trials_per_v = 1000
barrier = 2.5
ndt = 0.3
slope_A = 0.2
slope_B = 0.2
slope_C = 0.2
z_A = 0.5
z_B = 0.4
z_C = 0.6
params = {'A': {'a':barrier,'t':ndt,'v':slope_A,'z':z_A},
          'B': {'a':barrier,'t':ndt,'v':slope_B,'z':z_B},
          'C': {'a':barrier,'t':ndt,'v':slope_C,'z':z_C}
}
sim_data,sim_params = hddm.generate.gen_rand_data(params,size=trials_per_v,subjs=num_subs)
num_samples = 6000
num_burn = 1000
#model#
model_group4 = hddm.HDDM(sim_data,include='z',depends_on={'z':'condition'},informative=False)
model_group4.sample(num_samples,burn=num_burn)
model_group4_stats = model_group4.gen_stats()

###HDDM REGRESSION MODEL###
#link function#
def z_link_func(x, data=sim_data):
    return 1 / (1+np.exp(-x))
z_reg = {'model':'z ~ 1+ C(condition)','link_func': z_link_func}
reg = [z_reg]
model_group4_reg = hddm.HDDMRegressor(sim_data,reg,include='z',informative=False)   
model_group4_reg.sample(num_samples,burn=num_burn)
model_group4_reg_stats = model_group4_reg.gen_stats()

###calculate estimates for z###
z_Intercept = model_group4_reg_stats.loc[model_group4_reg_stats.index=='z_Intercept','mean']
z_TB = model_group4_reg_stats.loc[model_group4_reg_stats.index=='z_C(condition)[T.B]','mean']
z_TC = model_group4_reg_stats.loc[model_group4_reg_stats.index=='z_C(condition)[T.C]','mean']
z_A = 1/(1+np.exp(-(np.float(z_Intercept))))
z_B = 1/(1+np.exp(-(np.float(z_Intercept)+np.float(z_TB))))
z_C = 1/(1+np.exp(-(np.float(z_Intercept)+np.float(z_TC))))
